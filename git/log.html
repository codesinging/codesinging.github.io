# 历史 Log

# 查看当前文件变更状态

git status

# 显示当前分支的版本历史

git log

# 显示过去5次提交

git log -5 --pretty --oneline

# 显示commit历史，以及每次commit发生变更的文件

git log --stat

# 获取所有操作日志

git reflog

# 在一行内显示

git log --oneline

# 根据关键词搜索

git log --all --grep="commit"

# 查询某人的提交日志

git log --author="codesinging"

# 显示指定文件的修改历史

git blame <file>

# 显示所有提交过的用户，按提交次数排序

git shortlog -sn
